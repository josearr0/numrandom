Ejercicio donde el usuario adivina el numero:


public static void main(String[] args) {
Scanner sc= new Scanner(System.in);

int random = (int)Math.floor(Math.random()*(1000-1+1)+1);
int adivino;
do{
System.out.println("Ingresa un número: ");
adivino= sc.nextInt();
if (adivino>random) {
System.out.println("");
System.out.println("EL NUMERO ES MENOR QUE "+adivino);
} else if(adivino<random){
System.out.println("");
System.out.println("EL NUMERO ES MAYOR QUE "+ adivino);
}
System.out.println("");
}while(adivino!=random);
if (adivino==random) {
System.out.println("FELICIDADES! ADIVINASTE EL NUMERO: "+random);
}
}
}

Ejercicio donde el sistema adivina el numero:


public static void main(String[] args) {

int min=1;
int max=1000;
int resp=0;
Scanner sc= new Scanner(System.in);
int random =(int)Math.floor(Math.random()*(max-1+1)+min);
do{

System.out.println("¿Su número es "+ random+" ?");
System.out.println("1 (menor), 2 (mayor), 3 (sí)");
resp= sc.nextInt();

if (resp==1) {
max=random;
random = (int)Math.floor(Math.random()*(max-min+1)+min);
System.out.println(random);
} else if(resp==2){
min=random;
random = (int)Math.floor(Math.random()*(max-min+1)+min);
System.out.println(random);
} else if(resp==3){
break;
}
} while(resp!=3);
if (resp==3) {
System.out.println("Adivine el número: "+random);
}
}
